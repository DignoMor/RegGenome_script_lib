264    def _is_sorted(self) -> bool:
265        '''
266        Check if the table is sorted.
267        '''
268        for i in range(1, len(self._data_df)):
269            if self._data_df.iloc[i]["chrom"] < self._data_df.iloc[i-1]["chrom"]:
270                return False
271            if self._data_df.iloc[i]["chrom"] == self._data_df.iloc[i-1]["chrom"]:
272                if self._data_df.iloc[i]["start"] < self._data_df.iloc[i-1]["start"]:
273                    return False
274                if self._data_df.iloc[i]["start"] == self._data_df.iloc[i-1]["start"]:
275                    if self._data_df.iloc[i]["end"] < self._data_df.iloc[i-1]["end"]:
276                        return False
277
278        return True

508                                 chr2, start2, end2, 
509                                 column_name, 
510                                 overlapping_base1=1, 
511                                 overlapping_base2=1, 
512                                 ):
513        '''
514        Search for a pair end region and return the column data as indicated by the column name.
515        Return None if no data is found for the given pair regions.
516
517        Keyword arguments:
518        chr1 -- chromosome of the first region
519        start1 -- start of the first region
520        end1 -- end of the first region
521        chr2 -- chromosome of the second region
522        start2 -- start of the second region
523        end2 -- end of the second region
524        column_name -- the column name of the data to return
525        overlapping_base1 -- the number of overlapping bases required for a match for the first input region
526        overlapping_base2 -- the number of overlapping bases required for a match for the second input region
527        '''
528        # forward search
529        temp_ind_array = self.search_region(chr1, start1, end1, overlapping_base1)
530        temp_bed_table = self.subset_by_index(temp_ind_array)
531        forward_inds = temp_bed_table.search_second_region(chr2, start2, end2, overlapping_base2)
532        forward_result_bed_table = temp_bed_table.subset_by_index(forward_inds)
533
534        # backward search
535        temp_ind_array = self.search_region(chr2, start2, end2, overlapping_base2)
536        temp_bed_table = self.subset_by_index(temp_ind_array)
537        backward_inds = temp_bed_table.search_second_region(chr1, start1, end1, overlapping_base1)
538        backward_result_bed_table = temp_bed_table.subset_by_index(backward_inds)
539
540        all_result_bt = forward_result_bed_table.concat(backward_result_bed_table)
541
542        return all_result_bt.get_extra_column(column_name)
